## مصدر البيانات الوحيد

```javascript
// كل شي يُقرأ من Phase 0-5 — لا إدخال جديد
const payload = JSON.parse(localStorage.getItem('stratix_diagnostic_payload'));

// البيانات المتاحة:
const available = {
    // Phase 0: معلومات الشركة
    company: {
        name: payload.company.name,           // "مؤسسة الأمل"
        sector: payload.company.sector,       // "retail" → يحدد نوع التحليل
        employees: payload.company.employees, // 12 → يحدد MICRO vs SMALL
        age: payload.company.age,             // "3_years"
    },

    // Phase 1: البيانات المالية الوحيدة
    financial: {
        monthlyRevenue: payload.financial.monthlyRevenue,     // 45,000
        monthlyExpenses: payload.financial.monthlyExpenses,   // 32,000
        // ⚠️ هذا كل ما عندنا — لا نطلب أكثر من كذا
    },

    // Phase 2-5: السياق الاستراتيجي
    readiness: {
        financialSystems: payload.readiness.financialSystems, // 2/5
        reporting: payload.readiness.reporting,               // 1/5
    },
    strategy: {
        target: payload.strategy.target,                      // "product_expansion"
        investmentNeeded: payload.strategy.investmentNeeded,   // 100,000
    },
    risks: {
        financial: payload.risks.financial,                   // 4/5
    },

    // الحجم المحسوب
    level: payload.company.employees <= 10 ? 'MICRO' : 'SMALL',
};
```

## المقاييس الضمنية المحسوبة

```
┌─────────────────────────────────────────────────────────────────┐
│     المقاييس التي نحسبها من البيانات الموجودة                   │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  === MICRO + SMALL (مشترك) ===                                 │
│                                                                 │
│  🟢 الربح الشهري = الإيراد - المصروفات                          │
│     = 45,000 - 32,000 = 13,000 ﷼                               │
│                                                                 │
│  🟢 هامش الربح % = (الربح / الإيراد) × 100                     │
│     = (13,000 / 45,000) × 100 = 28.9%                          │
│                                                                 │
│  🟡 نقطة التعادل التقديرية = المصروفات الشهرية                  │
│     = 32,000 ﷼ (← تقدير — يحتاج تفصيل ثابت/متغير)             │
│                                                                 │
│  === SMALL فقط (11-49 موظف) ===                                │
│                                                                 │
│  🟡 Runway التقديري = (إيراد شهر × 3) / مصروفات               │
│     = (45K × 3) / 32K = 4.2 أشهر                              │
│     ← إذا توقف الإيراد، تعيش 4 أشهر                            │
│                                                                 │
│  🟡 قدرة التمويل الذاتي = ربح × 12 vs الاستثمار المطلوب       │
│     = 13K × 12 = 156K vs 100K المطلوبة                         │
│     ← تقدر تموّل التوسع ذاتياً خلال 8 أشهر                     │
│                                                                 │
│  🟡 نسبة المصروفات للإيراد = 71.1%                              │
│     ← Micro: مقبول | Small: يحتاج تحسين                        │
│                                                                 │
│  === الشفافية ===                                               │
│  محسوب بدقة:   الربح، الهامش                                   │
│  تقدير:        نقطة التعادل، Runway                             │
│  غير متوفر:    تفصيل ثابت/متغير، التدفق النقدي الفعلي          │
│  الثقة:        متوسطة (لأن البيانات محدودة)                     │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

## اكتشاف الأنماط — بالحجم

### أنماط MICRO (1-10 موظفين)

```
┌─────────────────────────────────────────────────────────────────┐
│              أنماط الشركات الصغيرة جداً                         │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  🔴 نمط 1: "المالك يفعل كل شي"                                 │
│  الشرط: employees ≤ 5 + financialSystems < 3                   │
│  الوصف: المالك هو المحاسب والمدير والبائع                      │
│  الأثر: قرارات عشوائية + إرهاق + خطر توقف                     │
│  الحل: محاسب خارجي جزئي (2-4K/شهر)                            │
│                                                                 │
│  🔴 نمط 2: "خلط الشخصي بالتجاري"                               │
│  الشرط: employees ≤ 5 + لا محاسب                               │
│  الوصف: حسابات شخصية = حسابات الشركة                           │
│  الأثر: لا يعرف ربحه الحقيقي                                   │
│  الحل: حساب بنكي منفصل + تطبيق محاسبة بسيط                    │
│                                                                 │
│  🟡 نمط 3: "عميل واحد = كل الإيراد"                             │
│  الشرط: client_concentration > 50%                             │
│  الوصف: إذا خسرت العميل الأكبر = كارثة                        │
│  الأثر: اعتماد خطير — لا مرونة تفاوضية                        │
│  الحل: تنويع قاعدة العملاء قبل أي شي                          │
│                                                                 │
│  🟢 نمط 4: "الهامش يكفي للبقاء"                                │
│  الشرط: profitMargin > 20%                                     │
│  الوصف: الشركة تعيش ← لكن هل تتطور؟                           │
│  السؤال: هل يكفي هذا الهامش لتمويل النمو؟                      │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### أنماط SMALL (11-49 موظف)

```
┌─────────────────────────────────────────────────────────────────┐
│              أنماط الشركات الصغيرة                              │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  🔴 نمط 1: "نمو بدون بنية"                                     │
│  الشرط: revenue > 30K + financialSystems < 3                   │

│  الأثر: البيانات غير موثوقة — قرارات استراتيجية خطرة           │
│  الحل: نظام محاسبة قبل أي توسع                                │
│                                                                 │
│  🔴 نمط 2: "فجوة تمويل التوسع"                                 │
│  الشرط: investmentNeeded > monthlyRevenue × 3                  │
│  الوصف: الاستثمار المطلوب يفوق قدرة السيولة                   │
│  الأثر: التوسع يحتاج تمويلاً خارجياً غير مؤمن                 │
│  السؤال: ما مصدر الـ 100K؟ قرض؟ شريك؟ تمويل ذكي؟              │
│                                                                 │
│  🟡 نمط 3: "تكاليف ثابتة عالية"                                │
│  الشرط: expenses/revenue > 75%                                 │
│  الوصف: المصروفات تأكل الإيراد — هامش ضيق                     │
│  الأثر: أي تراجع في المبيعات = خسارة فورية                    │
│  الحل: مراجعة هيكل التكاليف + تفاوض موردين                    │
│                                                                 │
│  🟡 نمط 4: "إيرادات غير مستقرة"                                │
│  الشرط: revenue_pattern = 'موسمية' أو 'غير منتظمة'            │
│  الوصف: لا يمكن التخطيط بدون إيراد منتظم                      │
│  الأثر: سيولة متذبذبة — مشاكل رواتب ومدفوعات                  │
│  الحل: احتياطي 3 أشهر + تنويع مصادر                           │
│                                                                 │
│  🟢 نمط 5: "ربحية صحية"                                        │
│  الشرط: profitMargin > 25% + revenue منتظم                     │
│  الوصف: الشركة مربحة ومستقرة                                  │
│  السؤال: هل الهامش مستدام أم مؤقت؟                             │
│                                                                 │
│  🟢 نمط 6: "تمويل ذاتي ممكن"                                   │
│  الشرط: (profit × 12) >= investmentNeeded                      │
│  الوصف: الشركة تقدر تموّل توسعها من أرباحها                   │
│  الأثر: إيجابي — لا حاجة لديون                                │
│  الجدول: investmentNeeded / monthlyProfit = عدد الأشهر          │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

## المنطق البرمجي المرجعي

```javascript
// === تحديد الحجم ===
function getLevel(employees) {
    return employees <= 10 ? 'MICRO' : 'SMALL';
}

// === حساب المقاييس الضمنية ===
function calculateImpliedMetrics(payload) {
    const f = payload.financial;
    const level = getLevel(payload.company.employees);
    
    const metrics = {
        // مشترك MICRO + SMALL
        monthlyProfit: f.monthlyRevenue - f.monthlyExpenses,
        profitMargin: ((f.monthlyRevenue - f.monthlyExpenses) / f.monthlyRevenue * 100),
        expenseRatio: (f.monthlyExpenses / f.monthlyRevenue * 100),
        breakEvenEstimate: f.monthlyExpenses, // تقدير
    };

    // SMALL فقط
    if (level === 'SMALL') {
        metrics.runway = (f.monthlyRevenue * 3) / f.monthlyExpenses;
        metrics.selfFundMonths = payload.strategy.investmentNeeded / metrics.monthlyProfit;
        metrics.canSelfFund = (metrics.monthlyProfit * 12) >= payload.strategy.investmentNeeded;
    }

    return metrics;
}

// === اكتشاف الأنماط ===
function detectPatterns(payload, metrics) {
    const level = getLevel(payload.company.employees);
    const patterns = [];

    if (level === 'MICRO') {
        if (payload.company.employees <= 5 && payload.readiness.financialSystems < 3) {
            patterns.push({ type:'critical', id:'owner_does_all',
                label:'المالك يفعل كل شي', action:'محاسب خارجي جزئي' });
        }
        if (metrics.profitMargin > 20) {
            patterns.push({ type:'positive', id:'margin_ok',
                label:'الهامش يكفي للبقاء', question:'هل يكفي للنمو؟' });
        }
    }

    if (level === 'SMALL') {
        if (f.monthlyRevenue > 30000 && payload.readiness.financialSystems < 3) {
            patterns.push({ type:'critical', id:'growth_no_infra',
                label:'نمو بدون بنية', action:'نظام محاسبة قبل التوسع' });
        }
        if (payload.strategy.investmentNeeded > f.monthlyRevenue * 3) {
            patterns.push({ type:'warning', id:'funding_gap',
                label:'فجوة تمويل التوسع', question:'ما مصدر التمويل؟' });
        }
        if (metrics.expenseRatio > 75) {
            patterns.push({ type:'warning', id:'high_costs',
                label:'تكاليف مرتفعة', action:'مراجعة هيكل التكاليف' });
        }
    }

    return patterns;
}

// === توليد الأسئلة بالحجم ===
function generateQuestions(payload) {
    const level = getLevel(payload.company.employees);
    
    if (level === 'MICRO') {
        return [
            { id:'account_separation', q:'هل تفصل حساباتك الشخصية عن التجارية؟', priority:'critical' },
            { id:'biggest_client_risk', q:'لو توقف أكبر عميل، كم شهر تصمد؟', priority:'critical' },
            { id:'real_profit', q:'هل تعرف كم ربحك الحقيقي؟', priority:'high' },
        ];
    }
    
    return [ // SMALL
        { id:'revenue_quality', q:'هل الإيرادات نقدية أم آجلة؟', priority:'critical' },
        { id:'cost_breakdown', q:'ما نسبة التكاليف الثابتة من المتغيرة؟', priority:'high' },
        { id:'expansion_funding', q:'هل مبلغ التوسع متوفر أم يحتاج تمويلاً؟', priority:'critical' },
        { id:'runway_actual', q:'كم شهر تمر بدون تمويل طارئ؟', priority:'high' },
        { id:'credit_line', q:'هل عندك خط ائتمان أو مصدر طوارئ؟', priority:'high' },
    ];
}
```

```
┌────────────────────────────────────────┐
│     ما يميز تحليل المالية هذا          │
├────────────────────────────────────────┤
│                                        │

│                                        │
│  🏢 يفرّق بالحجم                       │
```

## 🔒 ملاحظات الحفظ
- هذا التصور **محمي** ولا يُعدَّل بدون إذن صريح من صاحب المشروع
- البيانات (الإيرادات، المصروفات، عدد الموظفين) تتغير حسب كل عميل
- المنطق (الأنماط، الأسئلة، المعايير) **ثابت** لا يتغير
- التفريق MICRO vs SMALL **أساسي** لا يُلغى
- الكلمات المفتاحية: "تصور المالية" | "تصور المبيعات" | "تصور الامتثال"
